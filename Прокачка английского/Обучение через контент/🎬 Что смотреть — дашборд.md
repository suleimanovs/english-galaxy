# 🎬 Что смотреть — дашборд

> Контент-трекинг для прокачки английского. 🇺🇸 упор на американский акцент. Уровень и статус — в свойствах каждой карточки. Менять статус: открой тайтл → поле `status` (`watching` / `planned` / `watched` / `backlog`).

```dataviewjs
const SRC = "Прокачка английского/Обучение через контент";
const pages = dv.pages('"' + SRC + '"').where(p => p.type === "content");

const lvlColor = {A1:"#16a34a",A2:"#16a34a",B1:"#0891b2",B2:"#ea580c",C1:"#dc2626",C2:"#dc2626"};
const mediaRu  = {cartoon:"🎨 мультсериал",anime:"🌸 аниме (дубляж)",sitcom:"😄 ситком",drama:"🎭 сериал",movie:"🎬 фильм",learner:"📘 для изучающих"};
const lvlOrder = {A1:1,A2:2,B1:3,B2:4,C1:5,C2:6};
const esc = s => (s||"").toString().replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;");

function card(p){
  const lc = lvlColor[p.level] || "#64748b";
  const cover = p.cover
    ? `<img src="${esc(p.cover)}" loading="lazy" style="width:100%;aspect-ratio:2/3;object-fit:cover;display:block">`
    : `<div style="aspect-ratio:2/3;display:flex;align-items:center;justify-content:center;background:#1f2430;color:#7c8699;font-size:.78em;text-align:center;padding:8px">нет обложки<br>↳ поле cover:</div>`;
  const name = esc(p.file.name);
  const why = esc(p.why||"");
  return `<div style="border:1px solid var(--background-modifier-border);border-radius:10px;overflow:hidden;background:var(--background-secondary-alt);display:flex;flex-direction:column">
    <div style="position:relative">${cover}
      <span style="position:absolute;top:6px;right:6px;background:${lc};color:#fff;font-weight:700;font-size:.72em;padding:2px 7px;border-radius:20px">${esc(p.level)}</span></div>
    <div style="padding:8px 9px;display:flex;flex-direction:column;gap:3px">
      <a class="internal-link" data-href="${name}" href="${name}" style="font-weight:700;font-size:.9em;line-height:1.2">${esc(p.title)}</a>
      <div style="color:var(--text-muted);font-size:.71em">${mediaRu[p.media]||p.media} · ${p.year}</div>
      <div style="color:var(--text-muted);font-size:.73em;line-height:1.3">${why.length>92?why.slice(0,92)+"…":why}</div>
    </div></div>`;
}
const grid = arr => arr.length
  ? `<div style="display:grid;grid-template-columns:repeat(auto-fill,minmax(160px,1fr));gap:14px;margin:6px 0 20px">${arr.map(card).join("")}</div>`
  : "<p style='color:var(--text-muted);margin:4px 0 18px'>— пусто —</p>";
function section(title, arr){ dv.header(3, `${title} (${arr.length})`); dv.container.createDiv().innerHTML = grid(arr); }
const byStatus = s => pages.where(p=>p.status===s).sort(p=>lvlOrder[p.level]||9).array();

section("▶️ Сейчас смотрю",   byStatus("watching"));
section("⏭️ Дальше в очереди", byStatus("planned"));
section("✅ Просмотрено",      byStatus("watched"));

const backlog = pages.where(p=>p.status==="backlog").sort(p=>lvlOrder[p.level]||9);
dv.header(2, `📚 Каталог — что посмотреть дальше (${backlog.length})`);
dv.paragraph("От простого к сложному. Бери следующий тайтл на свой уровень и переводи в `planned`/`watching`.");
for(const lvl of ["A2","B1","B2","C1"]){
  const arr = backlog.where(p=>p.level===lvl).array();
  if(arr.length) section(`Уровень ${lvl}`, arr);
}
```

## 📋 Таблица (поиск/сортировка)
```dataview
TABLE WITHOUT ID
  ("[[" + file.name + "|" + title + "]]") as "Тайтл",
  level as "Ур.", upper(string(media)) as "Тип", status as "Статус", year as "Год"
FROM "Прокачка английского/Обучение через контент"
WHERE type = "content"
SORT level asc, status asc
```
